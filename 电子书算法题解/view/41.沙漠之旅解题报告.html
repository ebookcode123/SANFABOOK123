需要考虑各种情况


import java.io.*;
import java.util.*;

public class Solution {
    public static void main(String[] args) throws FileNotFoundException{
        Scanner in = new Scanner(new File("input.txt"));

        //read input process
        int n = in.nextInt();

        while(n!=-1){

            int tank = 0;
            int begin = -1;
            Queue<Integer> trip = new LinkedList<Integer>();
            for(int i=0;i<n;++i){

                int diff = in.nextInt() - in.nextInt();
                if(begin==-1){
                    if(diff>=0){
                        begin = i;
                        trip.add(diff);
                        tank = diff;
                    }
                }else{

                    if(tank+diff>=0){
                        trip.add(diff);
                        tank += diff;
                    }
                    else {

                        while (trip.isEmpty() == false && tank + diff < 0) {
                            int temp = trip.poll();
                            tank -= temp;
                            ++begin;
                        }

                        if (trip.isEmpty() == true) {
                            if (diff >= 0) {
                                begin = i;
                                trip.add(diff);
                                tank = diff;
                            } else {
                                begin = -1;
                            }
                        } else {
                            trip.add(diff);
                            tank += diff;
                        }
                    }
                }
            }
            System.out.println(begin);

            n =in.nextInt();
        }
        in.close();
    }
}

import java.io.*;
import java.util.*;

public class Solution {
    public static void main(String[] args) throws Exception {
        Scanner in = new Scanner(new File("input.txt"));

        int n = in.nextInt();

        while(n!=-1){
            int tank = 0;
            int begin = 0;
            int total = 0;

            for(int i=0;i<n;++i){
                int diff = in.nextInt() - in.nextInt(); // gas - distance
                total += diff;
                if(diff+tank>=0) tank += diff;
                else {
                    begin = i+1;
                    tank = 0;
                }
            }
            System.out.println(total>=0&&begin<n?begin:-1);

            n = in.nextInt();
        }
        in.close();
    }
}

如何证明total>=0，一定有解呢？

如果n＝1，自然成立

假设n<=k都成立

当n=k+1时，一定有一个节点diff>=0，说明这个节点可以到达下一个节点。 于是我们把这个节点和下一个节点捆绑在一起形成一个新的节点。此时成为k个节点。 于是得证。
