正面思考

基于沙漠之旅，我们已经理解了不断的构造，直到失败的方法。

这次虽然有了油箱，但是我们之前证明的一个结论依然是成立的：如果一个queue失败了，期间的任何节点都是失败的。

但是我们之前的另一个结论不成立了：如果总油量小于等于总距离，一定有解。因此我们需要循环更多以确定解是否存在。


import java.io.*;
import java.util.*;

public class Solution {
    public static void main(String[] args) throws Exception {
        Scanner in = new Scanner(new File("input.txt"));

        int n = in.nextInt(); // number of gas station
        while(n!=-1) {
            int m = in.nextInt(); // max gas volumn

            int[] gas = new int[n];
            int[] dis = new int[n];
            for(int i=0; i<n; ++i){
                gas[i] = in.nextInt();
                dis[i] = in.nextInt();
            }

            int trip = 0; // how long we have travelled
            int tank = 0; // our gas tank
            int begin = 0;

            for(int i=0;i<2*n;++i){
                int curr = i%n;
                tank += gas[curr]; // add gas in a station
                if(tank>m) tank = m;// check the max
                tank -= dis[curr]; // go to the next station
                if(tank<0){ //failed
                    begin = curr+1;
                    tank = 0;
                    trip = 0;
                }else{
                    if(++trip==n) break; // if I have gone through a round
                }
            }

            System.out.println(trip==n?begin:-1);

            n = in.nextInt();
        }
        in.close();
    }
}
