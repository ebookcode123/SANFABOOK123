Queue解法


import java.io.*;
import java.util.*;

class Generator{
    int cal(int n){
        if( n==0 || n==1 ) return n;
        int p2 = 1;
        int p3 = 1;
        int p5 = 1;

        int[] nums = new int[n+1];
        nums[1] = 1;

        int p = 1;
        while(p<n){
            int min = Math.min( nums[p2]*2, Math.min( nums[p3]*3, nums[p5]*5 ) );
            if(min == nums[p2]*2) ++p2;
            if(min == nums[p3]*3) ++p3;
            if(min == nums[p5]*5) ++p5;
            nums[++p]=min;
        }
        return nums[n];
    }

}

public class Solution {
    public static void main(String[] args) throws FileNotFoundException{
        Scanner in = new Scanner(new File("input.txt"));

        int n = in.nextInt(); // number of node

        Generator generator = new Generator();
        while(n!=-1){
            //build tree
            System.out.println(generator.cal(n));
            n = in.nextInt();
        }
        in.close();
    }
}
